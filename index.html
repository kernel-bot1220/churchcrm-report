💻 源码解读

关键代码片段与数据操作解析

📝 人员注册流程

// 1. 创建人员记录
$person = new Person();
$person->setFirstName($firstName);
$person->setLastName($lastName);
$person->setFamId($familyId);  // FK → family_fam
$person->setClsId($classification);  // 成员分类
$person->save();

// 2. 分配小组角色
$groupService->addUserToGroup($personId, $groupId, $roleId);
// → person2group2role_p2g2r 表插入记录

// 3. 活动签到
$eventAttend = new EventAttend();
$eventAttend->setEventId($eventId);
$eventAttend->setPersonId($personId);
$eventAttend->setCheckinDate(new DateTime());
// → event_attend 表记录

💰 奉献支付流程

// 1. 验证家庭
$family = FamilyQuery::create()->findOneById($famId);

// 2. 创建奉献记录 (Pledge or Payment)
$pledge = new Pledge();
$pledge->setFamId($famId);    // FK → family_fam
$pledge->setFundId($fundId);   // FK → donationfund_fun
$pledge->setAmount($amount);
$pledge->setSchedule($schedule);  // Weekly/Monthly/OneTime
$pledge->setPledgeOrPayment('Payment');  // 区分承诺和实际支付
$pledge->save();

// 3. 加入存款批次
$deposit = DepositQuery::create()->findOneById($depositId);
$deposit->addPledge($pledge);
$pledge->setDepId($depositId);  // FK → deposit_dep
$pledge->save();

// 4. 生成报税数据
$taxData = $financialService->getTaxReportData($fyId, $famId);

🔐 用户认证流程

// 1. 登录验证
$user = UserQuery::create()->findOneByUserName($username);
if (password_verify($password, $user->getPassword())) {
    // 2. 生成 Token → tokens 表
    $token = bin2hex(random_bytes(32));
    $tokenRecord = new Token();
    $tokenRecord->setToken($token);
    $tokenRecord->setType('api');
    $tokenRecord->setValidUntilDate($expiry);
    $tokenRecord->setReferenceId($user->getPerId());
    // 3. 返回 API Key 用于后续请求 Header
    return $user->getApiKey();
}
